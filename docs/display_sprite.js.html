const PIXI = require('../lib/pixi');

/**
 *
 * @memberof RGUI.Display
 */
class Sprite extends PIXI.Sprite {
  static fromBitmap (bitmap) {
    let sprite = Sprite.from(bitmap._canvas);
    sprite._bitmap = bitmap;
    return sprite
  }
}
